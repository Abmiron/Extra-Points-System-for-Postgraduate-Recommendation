server {
    listen 5173;
    server_name 112.124.59.110;
    index index.php index.html index.htm default.php default.htm default.html;
    root /www/wwwroot/gradpush/frontend/dist/;
    
    # ========== 解决双斜杠问题（保留以防万一） ==========
    rewrite ^//(.*)$ /$1 last;
    
    # ========== 处理SPA路由 ==========
    location / {
        try_files $uri $uri/ /index.html;
    
    }
    
    # ========== API代理配置 ==========
    # 代理所有/api开头的请求到后端
    location /api/ {
        # 移除/api前缀（如果后端不需要/api前缀）
        # rewrite ^/api/(.*)$ /$1 break;
        
        # 代理到后端服务（请确认后端端口）
        proxy_pass http://127.0.0.1:5001;  # 注意：这里的斜杠很重要！
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        
        # 超时设置
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;
        
        # CORS配置
        add_header Access-Control-Allow-Origin *;
        add_header Access-Control-Allow-Methods 'GET, POST, OPTIONS, PUT, DELETE';
        add_header Access-Control-Allow-Headers 'DNT,User-Agent,X-Requested-With,If-Modified-Since,Cache-Control,Content-Type,Range,Authorization';
        
        # 预检请求处理
        if ($request_method = 'OPTIONS') {
            add_header Access-Control-Allow-Origin *;
            add_header Access-Control-Allow-Methods 'GET, POST, OPTIONS, PUT, DELETE';
            add_header Access-Control-Allow-Headers 'DNT,User-Agent,X-Requested-With,If-Modified-Since,Cache-Control,Content-Type,Range,Authorization';
            add_header Access-Control-Max-Age 1728000;
            add_header Content-Type 'text/plain; charset=utf-8';
            add_header Content-Length 0;
            return 204;
        }
    }
    
    # ========== 文件下载配置 ==========
    # 代理所有/uploads开头的请求到后端，用于文件下载
    # 使用^~前缀提高优先级，确保uploads路径优先匹配
    location ^~ /uploads/ {
        # 代理到后端服务
        proxy_pass http://127.0.0.1:5001;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        
        # 超时设置（文件下载可能需要更长时间）
        proxy_connect_timeout 120s;
        proxy_send_timeout 120s;
        proxy_read_timeout 120s;
        
        # CORS配置
        add_header Access-Control-Allow-Origin *;
        add_header Access-Control-Allow-Methods 'GET, OPTIONS';
        add_header Access-Control-Allow-Headers 'DNT,User-Agent,X-Requested-With,If-Modified-Since,Cache-Control,Content-Type,Range,Authorization';
        
        # 预检请求处理
        if ($request_method = 'OPTIONS') {
            add_header Access-Control-Allow-Origin *;
            add_header Access-Control-Allow-Methods 'GET, OPTIONS';
            add_header Access-Control-Allow-Headers 'DNT,User-Agent,X-Requested-With,If-Modified-Since,Cache-Control,Content-Type,Range,Authorization';
            add_header Access-Control-Max-Age 1728000;
            add_header Content-Type 'text/plain; charset=utf-8';
            add_header Content-Length 0;
            return 204;
        }
    }
    
    # ========== 如果需要单独代理特定路径 ==========
    # 除了/api/路径，还有其他需要代理的路径
    location ~ ^/(applications|generate-captcha|faculties) {
        proxy_pass http://127.0.0.1:5001;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
    
    # ========== 宝塔面板原有配置 ==========
    #CERT-APPLY-CHECK--START
    include /www/server/panel/vhost/nginx/well-known/112.124.59.110.conf;
    #CERT-APPLY-CHECK--END
    include /www/server/panel/vhost/nginx/extension/112.124.59.110/*.conf;
    
    error_page 404 /404.html;
    
    location ~ /purge(/.*) {
        proxy_cache_purge cache_one $host$1$is_args$args;
    }
    
    include /www/server/panel/vhost/nginx/proxy/112.124.59.110/*.conf;
    include enable-php-82.conf;
    include /www/server/panel/vhost/rewrite/112.124.59.110.conf;
    
    location ~ ^/(\.user.ini|\.htaccess|\.git|\.env|\.svn|\.project|LICENSE|README.md) {
        return 404;
    }
    
    location ~ \.well-known {
        allow all;
    }
    
    if ( $uri ~ "^/\.well-known/.*\.(php|jsp|py|js|css|lua|ts|go|zip|tar\.gz|rar|7z|sql|bak)$" ) {
        return 403;
    }
    
    # 静态文件配置（注意：uploads目录已通过更高优先级的location处理）
    location ~ .*\.(gif|jpg|jpeg|png|bmp|swf)$
    {
        # 确保静态文件配置不会影响uploads目录
        if ($request_uri ~ ^/uploads/) {
            break;
        }
        expires      15d;
        error_log /dev/null;
        access_log /dev/null;
    }
    location ~ .*\.(js|css)?$
    {
        # 确保静态文件配置不会影响uploads目录
        if ($request_uri ~ ^/uploads/) {
            break;
        }
        expires      1h;
        error_log /dev/null;
        access_log /dev/null;
    }
	access_log  /www/wwwlogs/112.124.59.110.log;
    error_log  /www/wwwlogs/112.124.59.110.error.log;
}
